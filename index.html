    QUESTION NO1:-Find the Smaller Angle.
    Example
    Sample Input 1
    5 30

     Sample Output 1
     15

     Sample Input 2
     6 0

     Sample Output 2
     180
    
    ANSWER NO1:-
    var Minimal_Angle = (h, m) => 
     {
     h=h%12;
     let hour=(h*360)/12 + (m*360)/(12*60);
     let minute = (m*360)/60;
     let angle = Math.abs(hour-minute);
     if(angle>180){
     return (360-angle);
     }
     else{
       return angle;
     }
     };

    QUESTION NO3:-Perfect Number Check.
    Example
    Sample Input 1
    1
    
    Sample Output 1
    YES
    
    Sample Input 2
    96345
    
    Sample Output 2
    NO
    
    ANSWER NO3:-
    var Perfect_Check = (N) => 
    {
          //finding divisor & adding them
          let add =0;
          let x=1;
          //looping all divisor of the number
          while(N>=x){
            if(N%x===0){
              add=add+x;
            }
            x++;
          }
          //check if the addition of divisors is equal to 
          //the input number
          if(add===N)
            {
              return "YES";
            }
            else{
              return "NO";
            }
      };

    QUESTION NO4:-Reverse a Number.
    Example
    Sample Input 1
    1900
    
    Sample Output 1
    91
    
    Sample Input 2
    2012
    
    Sample Output 2
    2102
    
    ANSWER NO4:-
    var Reverse_Number = (N) => 
    {
      //initialize to store the number
      let revnum=0;
      //loop through the digits of the number and reverse them
      while(N>0){
        let r=N%10;
        revnum = revnum*10+r;
        N=Math.floor(N/10);
      }  
      return revnum;
    
    };
